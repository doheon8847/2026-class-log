#include <stdio.h>

int main()
{
    int num;
    while(1)           //참일때 반복 , 0->거짓(false) 1->참(true)
{                 //무한반복
    printf("숫자를 입력하세요: ");
    scanf("%d",&num);
    
    if(num==10) // num == 10
    {
        printf("정답입니다\n");
        break;
    }
    
    if(num > 10)   // 10보다 크면
    {
        printf("너무 높아요\n");
    }
    else if(num < 10)   // 10보다 작으면
    {
        printf("너무 낮아요\n");
    }
}
    
    return 0;
}
